```c++
for(int i=0; i<(int)_matrix.size(); i++) {
    for(int j=0; j<(int)_matrix[i].size(); j++) {
        if (_matrix[i][j]) printf("\x1b[41m");
        cout << ".";
        if (_matrix[i][j]) printf("\x1b[40m");
    }
    cout << endl;
}
```
